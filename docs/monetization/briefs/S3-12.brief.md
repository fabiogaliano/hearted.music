```typescript
export async function createAccountForBetterAuthUser(
  data: CreateBetterAuthAccountData,
): Promise<Result<Account, DbError>> {
  const supabase = createAdminSupabaseClient();

  const accountResult = await fromSupabaseSingle(
    supabase
      .from("account")
      .insert({
        better_auth_user_id: data.better_auth_user_id,
        email: data.email,
        display_name: data.display_name,
      })
      .select()
      .single(),
  );

  if (Result.isError(accountResult)) return accountResult;

  const account = accountResult.value;

  const billingInsert = env.BILLING_ENABLED
    ? { account_id: account.id }
    : {
        account_id: account.id,
        unlimited_access_source: "self_hosted" as const,
      };

  const { error: billingError } = await supabase
    .from("account_billing")
    .insert(billingInsert);

  if (billingError && billingError.code !== "23505") {
    return Result.err(
      new DatabaseError({
        code: billingError.code,
        message: `Failed to provision account_billing for account ${account.id}: ${billingError.message}`,
      }),
    );
  }

  if (!env.BILLING_ENABLED) {
    await supabase.rpc("reprioritize_pending_jobs_for_account", {
      p_account_id: account.id,
    });
  }

  return Result.ok(account);
}
```

**Current state:** S2-03 has been implemented. Every new account created via Better Auth gets an `account_billing` row. When `BILLING_ENABLED=false`, that row includes `unlimited_access_source = 'self_hosted'` and the reprioritization RPC is called. No changes needed.

```typescript
export async function selectEnrichmentWorkPlan(
  accountId: string,
  maxSongs: number,
): Promise<EnrichmentWorkPlan> {
  const supabase = createAdminSupabaseClient();

  const { data, error } = await supabase.rpc(
    "select_liked_song_ids_needing_enrichment_work",
    { p_account_id: accountId, p_limit: maxSongs },
  );

  if (error) {
    throw new Error(`Failed to select enrichment work plan: ${error.message}`);
  }

  const rows = data ?? [];

  const flags: SongStageFlags[] = rows.map((row) => ({
    songId: row.song_id,
    needsAudioFeatures: row.needs_audio_features,
    needsGenreTagging: row.needs_genre_tagging,
    needsAnalysis: row.needs_analysis,
    needsEmbedding: row.needs_embedding,
    needsContentActivation: row.needs_content_activation,
  }));

  return {
    allSongIds: flags.map((f) => f.songId),
    flags,
    needAudioFeatures: flags.filter((f) => f.needsAudioFeatures).map((f) => f.songId),
    needGenreTagging: flags.filter((f) => f.needsGenreTagging).map((f) => f.songId),
    needAnalysis: flags.filter((f) => f.needsAnalysis).map((f) => f.songId),
    needEmbedding: flags.filter((f) => f.needsEmbedding).map((f) => f.songId),
    needContentActivation: flags
      .filter((f) => f.needsContentActivation)
      .map((f) => f.songId),
  };
}
```

**Current state:** S3-01 is implemented. The orchestrator calls `selectEnrichmentWorkPlan` (not the old ungated selector).

### Content Activation (S3-03/S3-04 already completed)

**File:** `src/lib/workflows/enrichment-pipeline/stages/content-activation.ts`

```typescript
async function activateForSelfHosted(
  supabase: AdminSupabaseClient,
  accountId: string,
  songIds: string[],
): Promise<void> {
  await markItemsNew(accountId, "song", songIds);

  const { error } = await supabase.rpc("insert_song_unlocks_without_charge", {
    p_account_id: accountId,
    p_song_ids: songIds,
    p_source: "self_hosted",
  });

  if (error) {
    console.error(
      `[content-activation] self_hosted unlock failed for ${accountId}: ${error.message}`,
    );
  }
}

export async function runContentActivation(
  ctx: EnrichmentContext,
  songIds: string[],
): Promise<void> {
  if (songIds.length === 0) return;

  const supabase = createAdminSupabaseClient();
  const billingResult = await readBillingState(supabase, ctx.accountId);

  if (Result.isError(billingResult)) {
    console.error(
      `[content-activation] Failed to read billing state for ${ctx.accountId}: ${billingResult.error.message}`,
    );
    return;
  }

  const billing = billingResult.value;

  switch (billing.unlimitedAccess.kind) {
    case "subscription":
      await activateForUnlimitedSubscription(supabase, ctx.accountId, songIds);
      break;
    case "self_hosted":
      await activateForSelfHosted(supabase, ctx.accountId, songIds);
      break;
    case "none":
      await markItemsNew(ctx.accountId, "song", songIds);
      break;
  }
}
```

**Current state:** S3-03/S3-04 are implemented. Content activation correctly writes `item_status` rows AND creates unlock rows with `source='self_hosted'` for self_hosted accounts. No changes needed.

### Match Refresh Selector (S1-11 already completed)

```sql
CREATE OR REPLACE FUNCTION select_entitled_data_enriched_liked_song_ids(
  p_account_id UUID
)
RETURNS TABLE(song_id UUID)
LANGUAGE sql
STABLE
SECURITY DEFINER
SET search_path = public
AS $$
  WITH billing_facts AS (
    SELECT COALESCE(
      (
        SELECT
          unlimited_access_source IS NOT NULL
          AND (
            unlimited_access_source = 'self_hosted'
            OR (
              unlimited_access_source = 'subscription'
              AND subscription_status = 'active'
            )
          )
        FROM account_billing
        WHERE account_id = p_account_id
      ),
      false
    ) AS has_unlimited_access
  )
  SELECT ls.song_id
  FROM liked_song ls
  CROSS JOIN billing_facts bf
  INNER JOIN song s ON s.id = ls.song_id
  WHERE ls.account_id = p_account_id
    AND ls.unliked_at IS NULL
    AND EXISTS (SELECT 1 FROM song_audio_feature saf WHERE saf.song_id = ls.song_id)
    AND array_length(s.genres, 1) IS NOT NULL
    AND EXISTS (SELECT 1 FROM song_analysis sa WHERE sa.song_id = ls.song_id)
    AND EXISTS (SELECT 1 FROM song_embedding se WHERE se.song_id = ls.song_id)
    AND (
      bf.has_unlimited_access
      OR EXISTS (
        SELECT 1 FROM account_song_unlock asu
        WHERE asu.account_id = p_account_id
          AND asu.song_id = ls.song_id
          AND asu.revoked_at IS NULL
      )
    );
$$;
```

## Constraints (Frozen Rules)

From **DECISIONS.md §3, §5, §6** and **TERMINOLOGY.md §5**:

### Data Model Invariants

- **Every account gets an `account_billing` row in every deployment.** Missing row is an invariant violation, not a mode signal.
- **`self_hosted` is orthogonal to `plan`.** Those accounts may keep `plan='free'` and `credit_balance=0`.
- **SQL never infers deployment mode from missing rows or absent billing state.** The canonical entitlement predicate must be used everywhere: `unlimited_access_source IS NOT NULL AND (unlimited_access_source = 'self_hosted' OR (unlimited_access_source = 'subscription' AND subscription_status = 'active')) OR EXISTS account_song_unlock`.

### Workflow Invariants

- **Phase A (audio_features, genre_tagging) is ungated.** Returns for all liked songs regardless of entitlement.
- **Phase B (song_analysis) and Phase C (song_embedding) are gated by effective entitlement.** Only entitled songs have `true` for these flags.
- **Content activation writes `item_status` for all entitled + analyzed songs.** It also persists unlock rows with `source='self_hosted'` for self_hosted accounts.
- **Matching status is a sub-dimension of `analyzed` songs only.** Locked songs have `matching_status = NULL`.
- **Queue band:** `self_hosted` accounts get `priority` band (same as yearly unlimited).

### Terminology Frozen Values

- **Unlock source values:** `free_auto | pack | unlimited | self_hosted | admin`
- **Song display state:** `locked | pending | analyzing | analyzed | failed`
- **Unlimited access source:** `subscription | self_hosted` (NULL = none)
- **Queue bands:** `low | standard | priority`

## Open Questions / Risks

### 1. **No existing provider-disabled integration test**

There is no existing end-to-end integration test validating the full provider-disabled flow from account creation through enrichment to read models. This is what S3-12 must create. The story scope says "Integration test: full flow from provisioning to analyzed state" and "bun run test passes" — this requires either:
- A new `.integration.test.ts` file in `src/lib/workflows/enrichment-pipeline/__tests__/`
- Or a new test in `src/lib/domains/library/accounts/__tests__/` 
- With real or seeded database state (not just mocks)

Unit tests for individual components exist (content activation, orchestrator, billing queries), but they are isolated. No test currently validates the entire flow sequentially with billing-aware gates.

### 2. **Test infrastructure for provider-disabled mode**

The existing `analysis-pipeline-full-flow.integration.test.ts` uses a pattern of checking `FULL_FLOW_TEST=true` environment variable and `TEST_ACCOUNT_ID`. S3-12 may need to either:
- Use real database with a test account created on-the-fly
- Or implement a fixture/seed approach to prepare self_hosted account state
- The briefing does not specify which approach is intended

### 3. **Queue reprioritization after provisioning**

S2-03 calls `reprioritize_pending_jobs_for_account()` after setting `unlimited_access_source='self_hosted'`. This is a no-op at account creation time (no jobs exist yet). But the call does happen. The test must either:
- Mock the RPC call (if using mocks)
- Or verify the RPC is called correctly (if using real DB)

### 4. **Enrichment selector entitlement gates — per-song flags**

The `select_liked_song_ids_needing_enrichment_work` RPC has subtle entitlement logic:
- Phase A flags (`needs_audio_features`, `needs_genre_tagging`) are returned regardless of entitlement
- Phase B/C flags (`needs_analysis`, `needs_embedding`) require effective entitlement
- `needs_content_activation` also requires entitlement

For a self_hosted account with newly created unlock rows (via content activation), subsequent batch selections should show all songs as entitled. The test must verify this sequentially.

### 5. **Matching status nullification for locked songs**

S3-11 says "matching status is a sub-dimension of `analyzed` songs only". For a self_hosted account, all songs should eventually reach `analyzed` state (if enrichment completes), so `matching_status` should never be `NULL`. But the test should verify:
- Before analysis: song in `pending` state has `matching_status=NULL`
- After analysis: song in `analyzed` state can have matching status

The current implementation correctly nulls matching_status for locked songs (CASE WHEN NOT ent.is_entitled THEN NULL), but a self_hosted account never has locked songs (they're all entitled).

### 6. **Match snapshot refresh and entitled candidates**

The story says "Match refresh uses all enriched songs". The RPC `select_entitled_data_enriched_liked_song_ids()` is used by the match refresh orchestrator. For a self_hosted account, all 4-artifact songs should be included. The test must:
- Run enrichment to completion
- Verify `select_entitled_data_enriched_liked_song_ids()` returns all songs with audio_features + genres + analysis + embedding
- (Actually triggering the match refresh orchestrator may be out of scope for the validation test)

### 7. **No "regressions" baseline defined**

The acceptance criterion "No regressions from current ungated behavior" implies a specific baseline. The story scope says provider-disabled is "the fastest way to validate the app-side entitlement model" — it's a validation, not a feature addition. What is the "current ungated behavior" being validated against?
- Pre-billing, all songs were entitled by default
- Now, self_hosted accounts explicitly get `unlimited_access_source='self_hosted'`
- The test should verify that the end result is the same (all songs processed, all visible in read models)

The implementation already supports this, but the test strategy should make the "baseline" explicit.

## Summary of What Needs Testing

All Phase 3 enforcement stories (S3-01 through S3-11) have been implemented and unit-tested. S3-12 is a **validation story** that must create an **end-to-end integration test** verifying:

1. **Account creation:** Fresh `BILLING_ENABLED=false` account → `unlimited_access_source='self_hosted'` row
2. **Entitlement flow:** All songs pass through `selectEnrichmentWorkPlan()` → all get entitled flags → all complete enrichment
3. **Content activation:** All analyzed songs get `item_status` rows + `account_song_unlock` rows with `source='self_hosted'`
4. **Read models:** All songs show as `display_state='analyzed'` in `get_liked_songs_page()` (never `locked`)
5. **Match refresh:** All data-enriched songs appear in `select_entitled_data_enriched_liked_song_ids()`
6. **No regressions:** Behavior matches pre-billing full-library processing

The test should be **sequential and realistic** (not just component mocks), using either:
- A real test account in a test database
- Or a seeded/fixture approach with real RPC calls

This validates that the entire entitlement model works end-to-end and that `self_hosted` is truly equivalent to unlimited access.
