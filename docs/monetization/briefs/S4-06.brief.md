```ts
export const webhookRoutes = new Hono();

webhookRoutes.post("/webhooks/stripe", async (c) => {
	const signature = c.req.header("stripe-signature");
	if (!signature) {
		return c.json({ error: "Missing stripe-signature header" }, 401);
	}

	const rawBody = await c.req.raw.clone().text();

	let event: Stripe.Event;
	try {
		event = stripe.webhooks.constructEvent(
			rawBody,
			signature,
			env.STRIPE_WEBHOOK_SECRET,
		);
	} catch {
		return c.json({ error: "Invalid signature" }, 401);
	}

	const handler = EVENT_HANDLERS[event.type];
	if (!handler) {
		return c.json({ received: true });
	}

	let claimed: boolean;
	try {
		claimed = await claimWebhookEvent(event.id);
	} catch (err) {
		console.error(`Webhook event claim error for ${event.id}: ...`);
		return c.json({ error: "Internal error" }, 500);
	}

	if (!claimed) {
		return c.json({ received: true });
	}

	try {
		await handler(event);
	} catch (err) {
		const message = err instanceof Error ? err.message : "Unknown error";
		await markWebhookEvent(event.id, "failed", message);
		console.error(`Webhook handler error for ${event.type} (${event.id}): ${message}`);
	}

	return c.json({ received: true });
});
```

### Webhook Event Management Utilities

**File: `v1_hearted_brand/src/lib/webhook-event.ts`**

```ts
export async function claimWebhookEvent(stripeEventId: string): Promise<boolean> {
	const { count, error } = await supabase
		.from("billing_webhook_event")
		.insert({ stripe_event_id: stripeEventId, status: "processing" as const })
		.select("stripe_event_id")
		.single();

	if (error) {
		// 23505 = unique_violation (row already exists from a prior delivery)
		if (error.code === "23505") return false;
		throw new Error(`Failed to claim webhook event: ${error.message}`);
	}

	return true;
}

export async function markWebhookEvent(
	stripeEventId: string,
	status: Extract<WebhookEventStatus, "processed" | "failed">,
	errorMessage?: string,
): Promise<void> {
	const update: Record<string, unknown> = {
		status,
		updated_at: new Date().toISOString(),
	};
	if (status === "processed") {
		update.processed_at = new Date().toISOString();
	}
	if (errorMessage) {
		update.error_message = errorMessage.slice(0, 2000);
	}

	const { error } = await supabase
		.from("billing_webhook_event")
		.update(update)
		.eq("stripe_event_id", stripeEventId);

	if (error) {
		console.error(
			`Failed to update webhook event ${stripeEventId} to ${status}: ${error.message}`,
		);
	}
}
```

- **Idempotency pattern:** `INSERT ... ON CONFLICT DO NOTHING` via Supabase
- **Status progression:** `processing` → `processed` or `failed`

	stripe_event_id: string;

export async function sendBridgeEvent(payload: BridgePayload): Promise<Response> {
	const body = JSON.stringify(payload);
	const { timestamp, signature } = await signPayload(
		body,
		env.BILLING_SHARED_SECRET,
	);

	return fetch(`${env.APP_BRIDGE_URL}/api/billing/bridge`, {
		method: "POST",
		headers: {
			"Content-Type": "application/json",
			"X-Timestamp": timestamp,
			"X-Signature": signature,
		},
		body,
	});
}

### Pack Fulfillment Handler (Reference Pattern)

**File: `v1_hearted_brand/src/handlers/checkout-completed.ts`**

The `handlePackCheckout` function shows the pattern that S4-06 should follow for subscription handlers:

```ts
async function handlePackCheckout(
	stripeEventId: string,
	session: Stripe.Checkout.Session,
): Promise<void> {
	const accountId = session.metadata?.account_id;
	if (!accountId) {
		await markWebhookEvent(stripeEventId, "failed", "Missing account_id...");
		return;
	}

	// ... extract offer details from line items
	const priceCents = session.amount_total ?? 0;

	const { data, error } = await supabase.rpc("fulfill_pack_purchase", {
		p_account_id: accountId,
		p_stripe_event_id: stripeEventId,
		p_offer_id: offer.offer_id,
		p_credits: offer.credits,
		p_price_cents: priceCents,
	});

	if (error) {
		await markWebhookEvent(stripeEventId, "failed", `RPC error: ${error.message}`);
		return;
	}

	const result = data as { new_balance: number; bonus_unlocked_song_ids: string[] };

	try {
		const bridgeRes = await sendBridgeEvent({
			kind: "pack_fulfilled",
			stripe_event_id: stripeEventId,
			data: { /* ... */ },
		});

		if (!bridgeRes.ok) {
			const text = await bridgeRes.text().catch(() => "");
			await markWebhookEvent(
				stripeEventId,
				"failed",
				`Bridge call failed (${bridgeRes.status}): ${text.slice(0, 500)}`,
			);
			return;
		}
	} catch (err) {
		await markWebhookEvent(stripeEventId, "failed", `Bridge error: ...`);
		return;
	}

	await markWebhookEvent(stripeEventId, "processed");
}
```

**Key pattern:**
1. Extract `account_id` from event metadata
2. Call RPC with required parameters
3. Handle RPC error → mark failed
4. Call `sendBridgeEvent` with result
5. Handle bridge error → mark failed
6. Mark `processed` on success

### Subscription Checkout Handler (Reference)

**File: `v1_hearted_brand/src/handlers/checkout-completed.ts`**

Shows how subscription refs are stored without activation:

```ts
async function handleSubscriptionCheckout(
	stripeEventId: string,
	session: Stripe.Checkout.Session,
): Promise<void> {
	const accountId = session.metadata?.account_id;
	const subscriptionId =
		typeof session.subscription === "string"
			? session.subscription
			: session.subscription?.id;

	if (!accountId || !subscriptionId) {
		await markWebhookEvent(stripeEventId, "failed", "Missing account_id or subscription...");
		return;
	}

	const { error } = await supabase
		.from("account_billing")
		.update({
			stripe_subscription_id: subscriptionId,
			subscription_status: "incomplete",
			updated_at: new Date().toISOString(),
		})
		.eq("account_id", accountId);

	if (error) {
		await markWebhookEvent(stripeEventId, "failed", `Failed to store subscription refs: ...`);
		return;
	}

	await markWebhookEvent(stripeEventId, "processed");
}
```

**Note:** This stores refs only — no activation happens in S4-05. Activation is S4-06 scope via `invoice.paid` handler.

  p_subscription_period_end TIMESTAMPTZ

### Conversion Apply RPC (from S1-08)

**File: `supabase/migrations/20260405180000_conversion_lifecycle_rpcs.sql`**

`apply_subscription_upgrade_conversion(p_conversion_id, p_stripe_subscription_id, p_stripe_invoice_id, p_applied_stripe_event_id)`

This RPC is called by the `invoice.paid` handler for initial activation when `conversion_id` is present. It:
- Locks and validates conversion status is `pending`
- Consumes reserved credits from pack lots
- Reduces `account_billing.credit_balance`
- Writes credit transaction ledger
- Sets conversion status to `applied` and stores Stripe refs

## Constraints (Frozen)

### From DECISIONS.md

| Constraint | Details |
|---|---|
| **Bridge event kinds** | Frozen set: `pack_fulfilled \| unlimited_activated \| pack_reversed \| unlimited_period_reversed \| subscription_deactivated` |
| **Initial activation must apply conversion first** | If `conversion_id` present in subscription metadata, `apply_subscription_upgrade_conversion` is called before `activate_subscription` |
| **Renewals do NOT re-activate** | `invoice.paid` with `billing_reason=subscription_cycle` calls `update_subscription_state` only; does not call `activate_subscription` |
| **Deactivation does NOT restore converted pack value** | `deactivate_subscription` RPC already handles this; handlers just call it |
| **All handlers are idempotent** | Via `billing_webhook_event` idempotency: duplicate webhooks with same `stripe_event_id` are detected and skipped |
| **HMAC auth pattern for bridge calls** | Same as incoming app→service: `X-Timestamp` + `X-Signature` headers (signing already implemented in `bridge.ts`) |

## Open Questions / Risks

### 1. Invoice Object Structure and Billing Reason Extraction

**Issue:** The story mentions `billing_reason=subscription_create` vs `billing_reason=subscription_cycle`, but Stripe's `Invoice` object doesn't have a `billing_reason` field in the standard schema.

**Possible interpretations:**
- The distinction might be inferred from `subscription.metadata.conversion_id` presence (initial has it, renewal doesn't)
- Or the handler must detect it by checking `invoice.lines[0].subscription_item.billing_cycle_anchor` vs current period
- Or it's a custom metadata field added at subscription creation time

**Clarification needed:** How should the handler distinguish between initial activation (`invoice.paid`) and renewal (`invoice.paid`)?

### 2. Subscription Object Fetching

**Issue:** The `invoice.paid` webhook event contains an `invoice` object, but the Stripe subscription object may need to be fetched separately to get:
- `subscription.metadata.conversion_id`
- Current period end dates
- Current cancel-at-period-end state

**Question:** Should the handler call `stripe.subscriptions.retrieve(subscription_id)` to get fresh data, or is the `invoice.paid` event payload sufficient?

### 3. Period End Calculation for Renewal

**Issue:** For renewal (`invoice.paid` with `billing_reason=subscription_cycle`), the story says "refresh lifecycle state only" via `update_subscription_state`, but what values should be passed?

**Specifically:**
- `p_subscription_period_end` — should this be `invoice.period_end` or computed from subscription?
- Should other fields be updated, or just period_end?

### 4. Conversion Metadata Availability in Invoice Event

**Issue:** The `conversion_id` is stored in `subscription.metadata` (from checkout), but the `invoice.paid` event contains an `invoice` object, not a subscription object directly.

**Question:** Is the `conversion_id` accessible from the invoice's related subscription, or must the handler fetch the subscription separately?

**Current assumption:** Stripe's `invoice.subscription` field provides subscription ID; handler will call `stripe.subscriptions.retrieve()` to get metadata.

### 5. Bridge Event Schema for Unlimited Activation

**Issue:** The story specifies bridging to app with: `stripe_event_id`, `account_id`, `stripe_subscription_id`, `subscription_period_end`.

**Question:** Should additional fields be included in the bridge event's `data` object, such as:
- `plan` (quarterly/yearly)?
- `unlimited_access_source` ('subscription')?
- `subscription_status` ('active')?

**Current assumption:** Only the story-specified fields are sent.

### 6. Stripe Subscription Status Mapping

**Issue:** The `account_billing.subscription_status` column has frozen values: `none | active | past_due | canceled | unpaid | incomplete | incomplete_expired`. But Stripe subscription object has status field with values that may not align exactly.

**Mapping expected:**
- Stripe `status='active'` → store `'active'`
- Stripe `status='past_due'` → store `'past_due'`
- Stripe `status='canceled'` → store `'canceled'`
- Stripe `status='unpaid'` → store `'unpaid'`
- Stripe `status='incomplete'` → store `'incomplete'`
- Stripe `status='incomplete_expired'` → store `'incomplete_expired'`

**Assumption:** No transformation needed; values align directly.

### 7. Bridge Event Kind Extension

**Issue:** `BridgeEventKind` type in `bridge.ts` currently only includes `"pack_fulfilled"`.

**For S4-06, must extend to:**
```ts
type BridgeEventKind = "pack_fulfilled" | "unlimited_activated" | "subscription_deactivated";
```

**Note:** `pack_reversed` and `unlimited_period_reversed` are S4-07 scope (refund/chargeback handling).

### 8. Error Handling for Partially Applied Conversions

**Issue:** If `apply_subscription_upgrade_conversion` is called and succeeds, but then `activate_subscription` fails, the account will have a consumed conversion but no active subscription.

**Question:** Should the handler:
- Attempt to reverse the conversion on activation failure?
- Leave the conversion in `applied` state and mark the webhook as failed for manual recovery?
- Log and escalate separately?

### 9. Account Existence Check

**Issue:** Handlers will call RPCs assuming the `account_id` exists and has an `account_billing` row. If the account is missing, the RPC calls will silently do nothing (no rows match the WHERE clause).

**Question:** Should handlers validate account existence before calling RPCs, or is silent no-op acceptable?

### 10. Conversion ID Presence Check

**Issue:** For initial `invoice.paid`, the handler must check if `conversion_id` is present in subscription metadata to decide whether to call `apply_subscription_upgrade_conversion`.

**Current assumption:** If `conversion_id` is absent or null, skip the apply step and go straight to `activate_subscription`.

**Question:** Should missing `conversion_id` on an upgrade checkout (where conversion was prepared) be logged as a warning, or is it normal for some paths?

### 11. Clock Skew in Bridge Signing

**Issue:** The HMAC signature includes a timestamp at signing time. If webhook processing is delayed (queue backlog), the timestamp may be old.

**Risk:** The app's HMAC validation has a 5-minute clock skew window. Delayed webhooks may be rejected if processing takes > 5 minutes.

**Question:** Is this acceptable, or should timestamps be adjusted?

**Current assumption:** Immediate signing at handler time is correct; high-latency processing is out of scope.

### 12. Subscription Deleted Event and Subscription Fetch

**Issue:** When `customer.subscription.deleted` is received, the subscription object in the event may not contain full state (Stripe sometimes minimizes deleted event payloads).

**Question:** Should the handler:
- Fetch the full subscription object before deactivation?
- Or call `deactivate_subscription` with only the information available in the event?

The `deactivate_subscription` RPC doesn't require subscription state — it only updates `account_billing`. So no additional fetch is needed, but the handler should validate `account_id` from metadata.

## Summary

S4-06 implements five webhook event handlers that complete the subscription lifecycle. The webhook infrastructure from S4-05, lifecycle RPCs from S1-07, conversion RPCs from S1-08, and database schema are all in place. Handlers will follow the existing pattern from the pack fulfillment handler: extract metadata, call RPC, handle errors, send bridge event, mark processed.

**Key implementation scope:**
